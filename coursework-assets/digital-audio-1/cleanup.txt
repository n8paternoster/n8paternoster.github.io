function [ output ] = cleanup(x, userwidth)fftLen = 2^nextpow2(length(x));y = fft(x,fftLen);dft_Mag = abs(y);                     % The magnitude of the dftdft_Pha = angle(y);                   % The phase of the dftmaxvalue = max(dft_Mag);              % Finds the max valuemaxbin = find(dft_Mag==maxvalue);     % Finds the 2 indices (bins) of that valuehalfwidth = round(userwidth/2);y1index = maxbin(1)-halfwidth:maxbin(1)+halfwidth;  % Setting the two vectors y2index = maxbin(2)-halfwidth:maxbin(2)+halfwidth;  % according to user widthdft_Mag(y1index) = zeros(size(y1index)); % Those values in the two bindft_Mag(y2index) = zeros(size(y2index)); % vectors are set to 0z = dft_Mag.*exp(1i*dft_Pha);   % Reconstruction of the new magtnitude vector                                % with its phase to get a complex vectoro = ifft(z);    output = o;                            output = real(o);end
